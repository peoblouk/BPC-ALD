447static inline bool vector_iterator_is_inside_span(const struct TSpan *aSpan, const struct TVectorIterator *aIter)
448 {
449 if(aIter && aIter->iVector)
450 {
451 assert(aIter->iVector->iValues);
452 const TVectorElement *elem_ptr = aIter->iVector->iValues + aIter->iPos;
453
454 return (elem_ptr >= aSpan->iVector.iValues) && (elem_ptr < aSpan->iVector.iValues + aSpan->iVector.iSize);
455 }
456 return false;

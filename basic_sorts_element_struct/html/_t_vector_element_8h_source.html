73static inline int vector_element_comparator(const TVectorElement *aLeft, const TVectorElement *aRight)
74 {
75 assert(aLeft);
76 assert(aRight);
77
78 switch(compare_by)
79 {
80 case EByPoints:
81 if(aLeft->iPoints == aRight->iPoints)
82 return 0;
83 if(aLeft->iPoints < aRight->iPoints)
84 return -1;
85 return 1;
86 break; /* EByPoints */
87
88 case EById:
89 if(aLeft->iId == aRight->iId)
90 return 0;
91 if(aLeft->iId < aRight->iId)
92 return -1;
93 return 1;
94 break; /* EById */
95
96 case EByFirstName:
97 return strcmp(aLeft->iFirstName, aRight->iFirstName);
98 break; /* EByFirstName */
99
100 case EByLastName:
101 return strcmp(aLeft->iLastName, aRight->iLastName);
102 break; /* EByLastName */
103
104 default:
105 perror("Unknown TVectorElementCompareSelect value");
106 break;
107 }
108 return 1;
109 }

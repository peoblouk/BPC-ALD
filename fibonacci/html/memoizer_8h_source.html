87static inline TMemoizedFuncOutput memoizer_eval(struct TMemoizer *aM, TMemoizedFuncInput aInput)
88 {
89 assert(aM);
90 assert(aM->iArchive);
91 assert(aM->iMemoizedFunc);
92
93 for(size_t i = 0; i < aM->iSize; ++i)
94 if(aM->iArchive[i].iInput == aInput)
95 {
96 ++aM->iStats.iMemoizations;
97 return aM->iArchive[i].iOutput;
98 }
99
100 aM->iTmpOutput = aM->iMemoizedFunc(aInput);
101 ++aM->iStats.iCalculations;
102 aM->iPos %= aM->iCapacity;
103 aM->iArchive[aM->iPos] = (struct TMemoizerItem) { .iInput = aInput, .iOutput = aM->iTmpOutput };
104
105 if(aM->iSize < aM->iCapacity)
106 ++aM->iSize;
107
108 return aM->iArchive[aM->iPos++].iOutput;
109 }
